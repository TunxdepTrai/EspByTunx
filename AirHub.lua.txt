-- Tải Pepsi UI qua HttpGet (không dùng game:GetObjects)
local Library = loadstring(game:HttpGet("https://pastebin.com/raw/0zajXgzk"))() -- Link chứa mã Pepsi UI (đã reupload)

-- Tạo cửa sổ
local Window = Library:CreateWindow({
    Name = "Pepsi UI Demo",
    Themeable = {
        Image = "7059346386",
        Info = "Made by ChatGPT",
        Credit = false
    },
    Background = "",
    Theme = [[{"__Designer.Colors.background":"1E1237"}]]
})

-- Tạo tab
local MainTab = Window:CreateTab({ Name = "Main" })

-- Section bên trái
local LeftSection = MainTab:CreateSection({ Name = "Controls" })

LeftSection:AddToggle({
    Name = "Auto Aimbot",
    Value = false,
    Callback = function(val)
        print("Aimbot:", val)
    end
})

LeftSection:AddSlider({
    Name = "Sensitivity",
    Value = 0.3,
    Min = 0,
    Max = 1,
    Decimals = 2,
    Callback = function(val)
        print("Sensitivity:", val)
    end
})

-- Section bên phải
local RightSection = MainTab:CreateSection({
    Name = "Visual Settings",
    Side = "Right"
})

RightSection:AddColorpicker({
    Name = "ESP Color",
    Value = Color3.fromRGB(255, 0, 0),
    Callback = function(color)
        print("ESP Color:", color)
    end
})
